    const text = data.content?.map(b => b.text || '').join('') || '';
    if (text) {
      const divider = '\n\n' + '─'.repeat(44) + '\n\n';
      lastScript = lastScript + divider + text;
      document.getElementById('scriptOutput').textContent = lastScript;
    }
  } catch (err) {
    console.error(err);
  } finally {
    setExtrasDisabled(false);
    btn.innerHTML = orig;
  }
}

function setExtrasDisabled(val) {
  ['teaserBtn','hookBtn','linkedinBtn'].forEach(id => {
    document.getElementById(id).disabled = val;
  });
}

function copyScript() {
  if (!lastScript) return;
  navigator.clipboard.writeText(lastScript).then(() => {
    const btn = document.getElementById('copyBtn');
    btn.innerHTML = '<i class="ti ti-check" style="font-size:14px;"></i> <span class="copy-ok">Copied!</span>';
    setTimeout(() => {
      btn.innerHTML = '<i class="ti ti-copy" style="font-size:14px;"></i> Copy to clipboard';
    }, 2200);
  });
}

function clearForm() {
  ['vtype','sector','tone','duration','placement','cta'].forEach(id => document.getElementById(id).value = '');
  ['product','persona','value','pain','proof'].forEach(id => document.getElementById(id).value = '');
  document.getElementById('outputCard').style.display = 'none';
  document.getElementById('statusMsg').textContent = '';
  lastScript = '';
  lastBrief = '';
  window.scrollTo({ top: 0, behavior: 'smooth' });
}
